JavaScript에서 선택자

태그 선택자

Elements, Tag

document.getElementsByTagName(태그명)[순서];

document.getElementsByTagName('div')[0];

클래스 선택자

Elements, Class

document.getElementsByClassName(클래스명)[순서];

document.getElementsByClassName("class-name")[0];

아이디 선택자

Element, Id

document.getElementById(아이디명);

document.getElementById("idName");

간단한 선택자

querySelector, (Tag, Class, Id)

순서 지정 불가

document.querySelector(태그명, 클래스명, 아이디명 다 가능);

document.querySelector("div");

document.querySelector(".class-name");

document.querySelector("#idName");

간단한 전체 선택자

querySelectorAll, (Tag, Class, Id)

순서를 지정해주지 않을 경우 전체 선택

document.querySelectorAll(태그명, 클래스명, 아이디명 다 가능)[순서];

document.querySelectorAll("div")[0];

document.querySelectorAll(".class-name")[0];

document.querySelectorAll("#idName");
